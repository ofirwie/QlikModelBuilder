### StateManager Implementation:

```typescript
// src/panels/state/StateManager.ts

import * as vscode from 'vscode';
import { WizardState, DEFAULT_STATE } from './WizardState';

export class StateManager {
  private static readonly STATE_KEY = 'qmb.wizardState';
  private context: vscode.ExtensionContext;

  constructor(context: vscode.ExtensionContext) {
    this.context = context;
  }

  // Load state from storage
  async load(): Promise<WizardState> {
    const saved = this.context.workspaceState.get<WizardState>(
      StateManager.STATE_KEY
    );
    return saved || { ...DEFAULT_STATE };
  }

  // Save state to storage
  async save(state: WizardState): Promise<void> {
    await this.context.workspaceState.update(
      StateManager.STATE_KEY,
      state
    );
  }

  // Clear state (start fresh)
  async clear(): Promise<void> {
    await this.context.workspaceState.update(
      StateManager.STATE_KEY,
      undefined
    );
  }

  // Update single field
  async updateField<K extends keyof WizardState>(
    key: K,
    value: WizardState[K]
  ): Promise<WizardState> {
    const state = await this.load();
    state[key] = value;
    await this.save(state);
    return state;
  }
}
```

  private findSchemaSheet(sheetNames: string[]): string | null {
    const schemaNames = ['tables', '×˜×‘×œ××•×ª', 'schema', 'fields', '×©×“×•×ª'];

      schemaNames.includes(name.toLowerCase())

  private parseSchemaSheet(
    workbook: XLSX.WorkBook,
    sheetName: string,
    warnings: string[]
  ): TableDefinition[] {
    const sheet = workbook.Sheets[sheetName];
    const data = XLSX.utils.sheet_to_json<Record<string, any>>(sheet);

    const tablesMap = new Map<string, TableDefinition>();

    for (const row of data) {
      // ××¦× ××ª ×©× ×”×˜×‘×œ×” (×¢× aliases)

                        row['×©×_×˜×‘×œ×”'] || row.table;

                        row['×©×“×”'] || row['×¢××•×“×”'] || row.field;

        continue;

          fields: [],

          validation: { existsInSource: false, fieldsMatch: false, errors: [], warnings: [] }

      // ×”×•×¡×£ ×©×“×” (Raw Extract - ×œ×œ× type detection)

        include,
        rename: rename?.trim() || undefined

      if (isIncremental) {
        table.incremental.enabled = true;
        table.incremental.field = fieldName;

      }
    }

    return Array.from(tablesMap.values());

  private parseSheetsAsTables(
    workbook: XLSX.WorkBook,
    warnings: string[]
  ): TableDefinition[] {
    const tables: TableDefinition[] = [];
    const skipPatterns = ['readme', 'info', 'about', 'instructions'];

    for (const sheetName of workbook.SheetNames) {
      // ×“×œ×’ ×¢×œ sheets ×©×œ ××˜×-×“××˜×”
      if (sheetName.startsWith('_') ||
          skipPatterns.some(p => sheetName.toLowerCase().includes(p))) {
        warnings.push(`×“×™×œ×’×ª×™ ×¢×œ ×’×™×œ×™×•×Ÿ: ${sheetName}`);
        continue;
      }

      const sheet = workbook.Sheets[sheetName];
      const data = XLSX.utils.sheet_to_json(sheet, { header: 1 }) as any[][];

      if (data.length < 2) {
        warnings.push(`×’×™×œ×™×•×Ÿ "${sheetName}" ×¨×™×§ ××• ×¢× ×©×•×¨×” ××—×ª ×‘×œ×‘×“`);
        continue;
      }

      // ×©×•×¨×” ×¨××©×•× ×” = headers
      const headers = data[0] as string[];
      const validHeaders = headers.filter(h => h && String(h).trim());

      if (validHeaders.length === 0) {
        warnings.push(`×’×™×œ×™×•×Ÿ "${sheetName}" ×œ×œ× ×›×•×ª×¨×•×ª ×ª×§×™× ×•×ª`);
        continue;
      }

      // Raw Extract - ×¨×§ ×©××•×ª ×©×“×•×ª, ×œ×œ× type detection
      const fields = validHeaders.map((name) => ({
        name: String(name).trim(),
        include: true
        // ××™×Ÿ type - Qlik ×™×–×”×” ××•×˜×•××˜×™×ª
      }));

      // ×–×”×” ×©×“×” incremental
      const incrementalField = this.detectIncrementalField(fields);

      tables.push({
        name: sheetName,
        fields,
        incremental: {
          enabled: !!incrementalField,
          strategy: incrementalField ? 'insert_only' : 'full',
          field: incrementalField || '',
          keepHistory: !!incrementalField
        },
        validation: { existsInSource: false, fieldsMatch: false, errors: [], warnings: [] }
      });
    }

    return tables;
  }

  // âŒ detectType() × ××—×§×” - Raw Extract Philosophy
  // Qlik ×™×–×”×” ×˜×™×¤×•×¡×™× ××•×˜×•××˜×™×ª ×¢× Dual values

  // Raw Extract: ×–×™×”×•×™ ×©×“×” incremental ×œ×¤×™ ×©× ×‘×œ×‘×“ (×œ×œ× type)
  private detectIncrementalField(fields: FieldDefinition[]): string | null {
    const patterns = [
      /^modified/i, /^updated/i, /^lastmodified/i, /^lastupdated/i,
      /^modifydate/i, /^updatedate/i, /modifieddate$/i, /updateddate$/i,
      /timestamp$/i, /×ª××¨×™×š.*×¢×“×›×•×Ÿ/i, /×¢×“×›×•×Ÿ.*×ª××¨×™×š/i
    ];

    // ××—×¤×© ×œ×¤×™ ×©× ×”×©×“×” ×‘×œ×‘×“ - ×œ× ×œ×¤×™ type
    for (const field of fields) {
      if (patterns.some(p => p.test(field.name))) {
        return field.name;
      }
    }

    return null;
  }

  // âŒ normalizeType() × ××—×§×” - Raw Extract Philosophy
  // ×œ× ×¦×¨×™×š × ×¨××•×œ ×˜×™×¤×•×¡×™× ×‘×©×œ×‘ 1

  private parseBoolean(value: any): boolean {
    if (typeof value === 'boolean') return value;
    if (typeof value === 'string') {
      return ['yes', 'true', '1', '×›×Ÿ', 'v', 'x'].includes(value.toLowerCase());
    }
    return Boolean(value);
  }
}

import * as fs from 'fs';
import * as path from 'path';

export interface Template {
  id: string;
  name: string;
  description: string;
  icon: string;
  connectionType: string;
  suggestedTables: TablePattern[];
  defaultFields: FieldDefaults;
  scriptTemplate: ScriptTemplate;
}

export class TemplateManager {
  private templatesPath: string;
  private templates: Map<string, Template> = new Map();

  constructor(extensionPath: string) {
    this.templatesPath = path.join(
      extensionPath, 'src', 'services', 'templates', 'templates'
    );
    this.loadTemplates();
  }

  private loadTemplates(): void {
    const files = fs.readdirSync(this.templatesPath)
      .filter(f => f.endsWith('.json'));

    for (const file of files) {
      const content = fs.readFileSync(
        path.join(this.templatesPath, file), 'utf-8'
      );
      const template = JSON.parse(content) as Template;
      this.templates.set(template.id, template);
    }
  }

  list(): Template[] {
    return Array.from(this.templates.values());
  }

  get(id: string): Template | undefined {
    return this.templates.get(id);
  }

  // ×”×—×œ template ×¢×œ ×˜×‘×œ××•×ª
  applyToTables(
    templateId: string,
    tables: TableDefinition[]
  ): TableDefinition[] {
    const template = this.get(templateId);
    if (!template) return tables;

    return tables.map(table => {
      // ××¦× pattern ××ª××™×
      const pattern = template.suggestedTables.find(p =>
        new RegExp(p.pattern).test(table.name)
      );

      if (pattern) {
        // ×”×—×œ ×”×’×“×¨×•×ª incremental ××”-pattern
        table.incremental = {
          ...table.incremental,
          enabled: pattern.incremental.enabled,
          strategy: pattern.incremental.strategy as any
        };

        // ×—×¤×© ×©×“×” incremental ××•××œ×¥
        if (pattern.incremental.enabled && !table.incremental.field) {
          const suggestedField = table.fields.find(f =>
            pattern.incremental.suggestedFields.some(sf =>
              f.name.toLowerCase().includes(sf.toLowerCase())
            )
          );
          if (suggestedField) {
            table.incremental.field = suggestedField.name;
          }
        }
      }

      // ×”×¡×ª×¨ ×©×“×•×ª ×œ×¤×™ defaultFields.exclude
      table.fields = table.fields.map(field => ({
        ...field,
        include: !template.defaultFields.exclude.some(ex =>
          field.name.toLowerCase().includes(ex.toLowerCase())
        )
      }));

      return table;
    });
  }
}
